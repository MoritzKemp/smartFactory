@startuml

== Announcement & Bidding ==
Manager -> Worker : tell(taskAvailable)

alt couldn't achieve goal
	Worker -> Manager : tell(refuse)
else surely could achieve goal
	Worker -> Manager : tell(offer)
end

== Award ==
alt reject bid
	Manager -> Worker : tell(reject)
else accept bid
	Manager -> Worker : achieve(task)
end

== Expediting ==
alt failed to achive goal
	Worker -> Manager : tell(failure)
else successful accomplished
	Worker -> Manager : tell(finishTask)
end

@enduml
